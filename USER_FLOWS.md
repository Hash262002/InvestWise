// src/services/market-data.service.ts

// ============ STOCK SEARCH ============
async searchAssets(query: string): Promise<SearchResult[]> {
  const cacheKey = `search:${query.toLowerCase()}`;
  
  // Check cache
  const cached = await cache.get(cacheKey);
  if (cached) return cached;

  // Yahoo Finance Search API
  const response = await axios.get(
    'https://query1.finance.yahoo.com/v1/finance/search',
    {
      params: {
        q: query,
        quotesCount: 20,
        newsCount: 0,
        enableFuzzyQuery: true,
        quotesQueryId: 'tss_match_phrase_query',
      },
      headers: {
        'User-Agent': 'Mozilla/5.0',
      },
    }
  );

  const results = response.data.quotes
    .filter((q: any) => 
      q.exchange === 'NSE' || 
      q.exchange === 'BSE' || 
      q.quoteType === 'MUTUALFUND'
    )
    .map((q: any) => ({
      symbol: q.symbol,
      name: q.shortname || q.longname,
      type: q.quoteType,
      exchange: q.exchange,
    }));

  // Cache for 1 hour
  await cache.set(cacheKey, results, 3600);
  
  return results;
}

// ============ STOCK QUOTE ============
async getStockQuote(symbol: string): Promise<StockQuote> {
  const cacheKey = `quote:${symbol}`;
  
  const cached = await cache.get(cacheKey);
  if (cached) return cached;

  const response = await axios.get(
    `https://query1.finance.yahoo.com/v8/finance/chart/${symbol}`,
    {
      params: { interval: '1d', range: '1d' },
      headers: { 'User-Agent': 'Mozilla/5.0' },
    }
  );

  const meta = response.data.chart.result[0].meta;
  
  const quote = {
    symbol: meta.symbol,
    name: meta.shortName,
    price: meta.regularMarketPrice,
    previousClose: meta.previousClose,
    change: meta.regularMarketPrice - meta.previousClose,
    changePercent: ((meta.regularMarketPrice - meta.previousClose) / meta.previousClose) * 100,
    currency: meta.currency,
    exchange: meta.exchangeName,
  };

  // Cache for 30 seconds
  await cache.set(cacheKey, quote, 30);
  
  return quote;
}

// ============ MUTUAL FUND NAV ============
async getMutualFundNAV(schemeCode: string): Promise<MutualFundData> {
  const cacheKey = `mf:${schemeCode}`;
  
  const cached = await cache.get(cacheKey);
  if (cached) return cached;

  // Using MFAPI.in (FREE)
  const response = await axios.get(
    `https://api.mfapi.in/mf/${schemeCode}`
  );

  const data = {
    schemeCode: response.data.meta.scheme_code,
    schemeName: response.data.meta.scheme_name,
    fundHouse: response.data.meta.fund_house,
    schemeType: response.data.meta.scheme_type,
    nav: parseFloat(response.data.data[0].nav),
    date: response.data.data[0].date,
  };

  // Cache for 1 hour (MF NAV updates once daily)
  await cache.set(cacheKey, data, 3600);
  
  return data;
}

// ============ SEARCH MUTUAL FUNDS ============
async searchMutualFunds(query: string): Promise<MutualFundSearchResult[]> {
  // Get all schemes list (cached heavily)
  const schemesListKey = 'mf:all_schemes';
  let allSchemes = await cache.get(schemesListKey);
  
  if (!allSchemes) {
    const response = await axios.get('https://api.mfapi.in/mf');
    allSchemes = response.data;
    // Cache for 24 hours
    await cache.set(schemesListKey, allSchemes, 86400);
  }

  // Filter by query
  const results = allSchemes
    .filter((scheme: any) => 
      scheme.schemeName.toLowerCase().includes(query.toLowerCase())
    )
    .slice(0, 20);

  return results;
}

## Summary: Happy Path Flow

```
1. USER REGISTERS
   └── Creates account with email, password, risk tolerance
   └── Gets JWT tokens, redirected to home

2. USER LANDS ON HOME PAGE
   └── Sees stats (0 portfolios initially)
   └── Sees prominent "Create Portfolio" button

3. USER CREATES PORTFOLIO
   └── Clicks "Create Portfolio"
   └── Enters name: "My Long Term Investments"
   └── Enters description (optional)
   └── Clicks "Next"

4. USER ADDS HOLDINGS
   └── Searches "Birla" → Finds GRASIM.NS
   └── Clicks "Add" → Modal opens
   └── Enters: 20 shares @ ₹2,300
   └── Clicks "Add to Portfolio"
   
   └── Searches "Tata Motors" → Finds TATAMOTORS.NS
   └── Adds: 10 shares @ ₹720
   
   └── Searches "ONGC" → Finds ONGC.NS
   └── Adds: 5 shares @ ₹260
   
   └── Clicks "Save Portfolio"

5. USER VIEWS PORTFOLIO
   └── Sees all 3 holdings with current prices
   └── Sees total value, gains/losses
   └── Sees "Analyze Portfolio" button

6. USER TRIGGERS AI ANALYSIS
   └── Clicks "Analyze Portfolio with AI"
   └── Sees loading state with agent progress
   └── Waits 30-60 seconds

7. USER VIEWS RESULTS
   └── Sees overall risk score: 42/100 (Medium)
   └── Sees GRASIM.NS: 🟢 Safe (28/100)
   └── Sees TATAMOTORS.NS: 🟢 Safe (35/100)
   └── Sees ONGC.NS: 🔴 Risky (72/100)
   └── Sees AI recommendations
   └── Can download report or re-analyze
```
